def test(model, key, params):
    score_results = {}
    true_results = {}
    dist_results = {}
    rmse_results = {}

    for i in range(0, 100):
        
        score_test = []
        true_beta = []
        dist_beta = []
        rmse_test = []

        X = datasets[i]
        Y = weights[i]

        modelcv = {key: GridSearchCV(model, param_grid=params).fit(X, Y).best_estimator_}

        for idxtrain, idxtest in kf.split(X):
            xtrain = X[idxtrain,:]
            ytrain = Y[idxtrain]
            xtest  = X[idxtest,:]
            ytest  = Y[idxtest]

            beta_hat = modelcv[key].coef_
            pos_model = np.where(beta_hat!=0)

            modelcv[key].fit(xtrain, ytrain)

            score_test.append(r2(ytest, modelcv[key].predict(xtest)))
            true_beta.append(np.intersect1d(pos, pos_model).shape[0])
            dist_beta.append(np.linalg.norm(beta_hat-beta_star, ord=2))
            rmse_test.append(np.sqrt(mse(ytest, modelcv[key].predict(xtest))))

            score_test = [np.mean(score_test)]
            true_beta = [np.mean(true_beta)]
            dist_beta = [np.mean(dist_beta)]
            rmse_test = [np.mean(rmse_test)]
        
        for k in score_test:
            score_results.setdefault('R^2', [])
            score_results['R^2'].append(k)

        for l in true_beta:
            true_results.setdefault('True Intersections', [])
            true_results['True Intersections'].append(l)

        for m in dist_beta:
            dist_results.setdefault('Euclidean Distance', [])
            dist_results['Euclidean Distance'].append(m)

        for n in rmse_test:
            rmse_results.setdefault('Root MSE', [])
            rmse_results['Root MSE'].append(n)
    
    return score_results, true_results, dist_results, rmse_results

Square Root Lasso

# Define our SKLearn compliant Square Root Lasso function
class SqRtLasso(BaseEstimator, RegressorMixin):
    def __init__(self, max_iter=50000, alpha=0.01):
        self.max_iter, self.alpha = max_iter, alpha
   
    def fit(self, x, y):
        alpha = self.alpha
        max_iter = self.max_iter

        def f_obj(x, y, beta, alpha):
            n =len(x)
            output = np.sqrt(1/n*np.sum((y-x.dot(beta))**2)) + alpha*np.sum(np.abs(beta))
            return output

        def f_grad(x, y, beta, alpha):
            n = x.shape[0]
            p = x.shape[1]
            beta = beta.flatten()
            beta = beta.reshape(-1,1)
            output = np.array((-1/np.sqrt(n))*np.transpose(x).dot(y-x.dot(beta))/np.sqrt(np.sum((y-x.dot(beta))**2))+alpha*np.sign(beta)).flatten()
            
            return output

        def objective(beta):
            return(f_obj(x, y, beta, alpha))
      
        def gradient(beta):
            return(f_grad(x, y, beta, alpha))
        
        beta0 = np.ones((x.shape[1], 1))
        output = minimize(objective, beta0, method='L-BFGS-B', jac=gradient, options={'gtol':1e-8,'max_iter':self.max_iter,'maxls':25,'disp':True})
        beta = output.x
        self.coef_ = beta
        
    def predict(self, x):
        return x.dot(self.coef_)

SCAD

# Define our SKLearn compliant smoothly clipped absolute deviation function
class SCAD(BaseEstimator, RegressorMixin):
    def __init__(self, max_iter=1000000, llambda=0.1, alpha=0.1):
        self.max_iter, self.llambda, self.alpha = max_iter, llambda, alpha
    
    def fit(self, x, y):
        alpha = self.alpha
        llambda = self.llambda
        max_iter = self.max_iter

        def scad_model(x, y, llambda, alpha):
            n = x.shape[0]
            p = x.shape[1]

        def scad_penalty(beta_hat, llambda, alpha):
            is_linear = (np.abs(beta_hat) <= llambda)
            is_quadratic = np.logical_and(llambda < np.abs(beta_hat), np.abs(beta_hat) <= alpha * llambda)
            is_constant = (alpha * llambda) < np.abs(beta_hat)

            linear_part = llambda * np.abs(beta_hat) * is_linear
            quadratic_part = (2 * alpha * llambda * np.abs(beta_hat) - beta_hat**2 - llambda**2) / (2 * (alpha - 1)) * is_quadratic
            constant_part = (llambda**2 * (alpha + 1)) / 2 * is_constant
            
            return linear_part + quadratic_part + constant_part

        def scad_derivative(beta_hat, llambda, alpha):
            return llambda * ((beta_hat <= llambda) + (alpha * llambda - beta_hat)*((alpha * llambda - beta_hat) > 0) / ((alpha - 1) * llambda) * (beta_hat > llambda))

        def scad(beta):
            beta = beta.flatten()
            beta = beta.reshape(-1,1)
            n = len(y)
            return 1/n*np.sum((y-x.dot(beta))**2) + np.sum(scad_penalty(beta, llambda, alpha))

        def dscad(beta):
            beta = beta.flatten()
            beta = beta.reshape(-1,1)
            n = len(y)
            return np.array(-2/n*np.transpose(x).dot(y-x.dot(beta))+scad_derivative(beta, llambda, alpha)).flatten()

        beta0 = np.ones((x.shape[1], 1))
        output = minimize(scad, beta0, method='L-BFGS-B', jac=dscad, options={'gtol':1e-8,'max_iter':max_iter,'maxls':25,'disp':True})
        beta = output.x
        self.coef_ = beta

    def predict(self, x):
        return x.dot(self.coef_)
